How Quick Sort Works:

1. Select a pivot element (usually the rightmost element)

2. Partition the array: elements < pivot go to left, elements> pivot go to right

3. Recursively sort the left and right subarrays

4. The i pointer tracks the boundary between elements less than the pivot and elements greater than the pivot

5. The j pointer scans through the array comparing elements with the pivot

Quick Sort Algorithm

function quickSort(arr, low, high) {
    if (low < high) {
        let pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

function partition(arr, low, high) {
    let pivot = arr[high];
    let i = low - 1;
    for (let j = low; j < high; j++) {
        if (arr[j] < pivot) {
            i++;
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
    }
    [arr[i + 1], arr[high]] = [arr[high], arr[i + 1]];
    return i + 1;
}
